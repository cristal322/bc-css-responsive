⚙️ Configuración de Producción Optimizada

Configuración CSS Optimizada para Producción 

/* Configuración de producción optimizada */
@import "tailwindcss";

@theme {
  /* Variables optimizadas para runtime performance */
  --color-primary-50: oklch(0.97 0.013 250.84);
  --color-primary-500: oklch(0.656 0.182 255.85);
  --color-primary-600: oklch(0.599 0.177 255.85);
  
  /* Fuentes con display optimizado */
  --font-sans: ui-sans-serif, system-ui, sans-serif;
  --font-display: "Inter", var(--font-sans);
  
  /* Espaciado con escala modular optimizada */
  --spacing-xs: 0.5rem;
  --spacing-sm: 0.75rem;
  --spacing-md: 1rem;
  --spacing-lg: 1.5rem;
  --spacing-xl: 2rem;
  
  /* Breakpoints optimizados para contenido */
  --breakpoint-xs: 20rem;    /* 320px - Mobile small */
  --breakpoint-sm: 24rem;    /* 384px - Mobile */  
  --breakpoint-md: 28rem;    /* 448px - Mobile large */
  --breakpoint-lg: 32rem;    /* 512px - Tablet small */
  --breakpoint-xl: 42rem;    /* 672px - Tablet */
  --breakpoint-2xl: 56rem;   /* 896px - Desktop */
  
  /* Animaciones optimizadas para performance */
  --duration-fast: 150ms;
  --duration-normal: 250ms;
  --duration-slow: 350ms;
  
  --ease-in: cubic-bezier(0.4, 0, 1, 1);
  --ease-out: cubic-bezier(0, 0, 0.2, 1);
  --ease-in-out: cubic-bezier(0.4, 0, 0.2, 1);
}

/* Componentes base optimizados */
@layer base {
  /* Optimización de fonts loading */
  @font-face {
    font-family: 'Inter';
    font-style: normal;
    font-weight: 400;
    font-display: swap; /* Mejora LCP */
    src: local('Inter'), url('/fonts/inter.woff2') format('woff2');
  }
  
  /* CSS Reset optimizado */
  *,
  ::before,
  ::after {
    box-sizing: border-box;
    border-width: 0;
    border-style: solid;
    border-color: theme(colors.gray.200);
  }
  
  html {
    line-height: 1.5;
    -webkit-text-size-adjust: 100%;
    tab-size: 4;
  }
  
  body {
    margin: 0;
    font-family: theme(fontFamily.sans);
    font-feature-settings: "kern";
  }
}

/* Componentes personalizados con performance */
@layer components {
  .btn {
    @apply inline-flex items-center justify-center px-4 py-2 
           border border-transparent rounded-lg font-medium 
           focus:outline-none focus:ring-2 focus:ring-offset-2
           transition-all duration-150 ease-out;
  }
  
  .btn-primary {
    @apply btn bg-primary-500 text-white 
           hover:bg-primary-600 focus:ring-primary-500
           shadow-sm hover:shadow-md;
  }
  
  .card {
    @apply bg-white border border-gray-200 rounded-xl 
           shadow-sm hover:shadow-md transition-shadow duration-200;
  }
  
  /* Layout component optimizado */
  .container-fluid {
    @apply w-full max-w-none px-4 sm:px-6 lg:px-8;
  }
  
  .container {
    @apply w-full max-w-7xl mx-auto px-4 sm:px-6 lg:px-8;
  }
}

/* Utilidades de performance */
@layer utilities {
  /* GPU acceleration para animaciones */
  .will-change-transform {
    will-change: transform;
  }
  
  .will-change-auto {
    will-change: auto;
  }
  
  /* Optimización de images */
  .img-responsive {
    @apply max-w-full h-auto;
    image-rendering: -webkit-optimize-contrast;
  }
  
  /* Text rendering optimizations */
  .text-render-optimized {
    text-rendering: optimizeLegibility;
    -webkit-font-smoothing: antialiased;
    -moz-osx-font-smoothing: grayscale;
  }
  
  /* Scroll optimizations */
  .scroll-smooth {
    scroll-behavior: smooth;
  }
  
  .overscroll-none {
    overscroll-behavior: none;
  }
}

/* Media queries para optimización específica */
@media (prefers-reduced-motion: reduce) {
  *,
  ::before,
  ::after {
    animation-duration: 0.01ms !important;
    animation-iteration-count: 1 !important;
    transition-duration: 0.01ms !important;
    scroll-behavior: auto !important;
  }
}

/* Optimización para impresión */
@media print {
  .no-print {
    display: none !important;
  }
  
  .print-force-break {
    page-break-after: always;
  }
}
🎯 Puntos Clave de Optimización
  • • Variables OKLCH: Mejor gamut de color y performance
  • • Font-display: swap: Mejora LCP (Largest Contentful Paint)
  • • Will-change: Optimización GPU para animaciones
  • • Prefers-reduced-motion: Accesibilidad y performance
  • • Sistema de breakpoints: Basado en contenido, no dispositivos

Configuración de Build para Máximo Performance

📦 Package.json Scripts
{
  "scripts": {
    "dev": "tailwindcss -i ./src/input.css -o ./dist/output.css --watch",
    "build": "npm run build:css && npm run optimize",
    "build:css": "tailwindcss -i ./src/input.css -o ./dist/output.css --minify",
    "optimize": "npm run purge:css && npm run compress",
    "purge:css": "purgecss --css ./dist/output.css --content './src/**/*.html' './src/**/*.js' --output ./dist/",
    "compress": "gzip -9 ./dist/output.css",
    "analyze": "tailwindcss-analyzer ./dist/output.css",
    "critical": "critical src/index.html --base dist/ --css dist/output.css --target critical.css --width 1300 --height 900"
  },
  "devDependencies": {
    "@tailwindcss/cli": "^4.0.0-alpha.1",
    "purgecss": "^5.0.0",
    "critical": "^6.0.0",
    "tailwindcss-analyzer": "^1.0.0"
  }
}
🚀 Webpack Integration
const path = require('path');
const MiniCssExtractPlugin = require('mini-css-extract-plugin');
const CssMinimizerPlugin = require('css-minimizer-webpack-plugin');

module.exports = {
  mode: 'production',
  entry: './src/index.js',
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: '[name].[contenthash].js',
    clean: true,
  },
  module: {
    rules: [
      {
        test: /\.css$/i,
        use: [
          MiniCssExtractPlugin.loader,
          'css-loader',
          {
            loader: 'postcss-loader',
            options: {
              postcssOptions: {
                plugins: [
                  require('@tailwindcss/postcss'),
                  require('autoprefixer'),
                  require('cssnano')({
                    preset: ['advanced', {
                      discardComments: { removeAll: true },
                      normalizeWhitespace: false,
                    }]
                  })
                ],
              },
            },
          },
        ],
      },
    ],
  },
  plugins: [
    new MiniCssExtractPlugin({
      filename: '[name].[contenthash].css',
    }),
  ],
  optimization: {
    minimizer: [
      '...',
      new CssMinimizerPlugin({
        minimizerOptions: {
          preset: [
            'advanced',
            {
              discardComments: { removeAll: true },
            },
          ],
        },
      }),
    ],
    splitChunks: {
      cacheGroups: {
        styles: {
          name: 'styles',
          type: 'css/mini-extract',
          chunks: 'all',
          enforce: true,
        },
      },
    },
  },
};

📁 Organización de Código

├── styles/
│   ├── input.css              # Configuración principal
│   ├── components/
│   │   ├── buttons.css        # Componentes específicos
│   │   ├── forms.css
│   │   └── navigation.css
│   └── utilities/
│       ├── animations.css     # Utilidades custom
│       └── helpers.css
├── components/
│   ├── ui/                    # Componentes reutilizables
│   │   ├── Button.jsx
│   │   ├── Card.jsx
│   │   └── Modal.jsx
│   └── layout/
│       ├── Header.jsx
│       └── Sidebar.jsx
└── pages/
    ├── Home.jsx
    └── Dashboard.jsx
📋 Naming Conventions
  • • Prefijos para components: .btn-*, .card-*
  • • States descriptivos: .is-loading, .has-error
  • • Variants claros: .btn-primary-lg, .text-heading-xl
  • • Utilities específicas: .layout-*, .animate-*
